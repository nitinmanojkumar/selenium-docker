pipeline {
    agent any
    stages {
        stage('Build Jar') {
            steps {
                bat "cd C://selenium-docker-master && mvn clean package -DskipTests"
            }
            
        }
                
        stage('Build Image') {
            steps {
                bat "cd C://selenium-docker-master && docker build -t=nitinmanojkumar/selenium-docker ."
            }
            
        }

        stage('Push Image') {
            steps {
                withCredentials([usernamePassword(credentialsId: 'dockerhub' , passwordVariable: 'pass', usernameVariable: 'user')]){
                bat "docker login --username=${user} --password=${pass}"
                bat "docker push nitinmanojkumar/selenium-docker:latest"
                }
            }
        }
    }
}
